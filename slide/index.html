✅ 実装した解決策


export async function getBuildings(client, args) {
  const { output_path } = args;
  
  if (output_path) {
    // 🎯 ファイル保存モード
    const result = await executeGeoJSONQuery(client, query, output_path);
    
    return {
      content: [{
        type: 'text',
        text: JSON.stringify({
          status: 'success',
          message: '建物データをダウンロードしました',
          file: output_path,           // ファイルパス
          size: result.size,           // "2.3 MB"
          feature_count: result.feature_count  // 1,234件
        }, null, 2)
      }]
    };
  }
  // 小さなデータは従来通り直接返却
}

実装詳細：ストリーミングダウンロード

コア機能：utils/file-downloader.js


// ストリーミングダウンロードの実装
export async function downloadToFile(url, query, outputPath) {
  return new Promise((resolve, reject) => {
    const req = https.request(options, (res) => {
      const writeStream = createWriteStream(outputPath);
      let downloadedBytes = 0;

      res.on('data', (chunk) => {
        downloadedBytes += chunk.length;
        // 進行状況を表示
        if (downloadedBytes % (1024 * 1024) === 0) {
          console.error(`Downloaded: ${(downloadedBytes/1024/1024).toFixed(1)} MB`);
        }
      });

      res.pipe(writeStream);  // ストリーミング書き込み
      
      writeStream.on('finish', () => {
        resolve({ size: `${(downloadedBytes/1024/1024).toFixed(2)} MB` });
      });
    });
  });
}

GeoJSON変換+保存：executeGeoJSONQuery


export async function executeGeoJSONQuery(overpassClient, query, outputPath) {
  // 1. OSMデータを一時ファイルにダウンロード
  const tempPath = outputPath + '.tmp.osm.json';
  await downloadToFile(server.url, fullQuery, tempPath);
  
  // 2. OSMデータを読み込み
  const osmData = JSON.parse(await fs.readFile(tempPath));
  
  // 3. GeoJSONに変換
  const geojson = osmtogeojson(osmData);
  
  // 4. 最終ファイルに保存
  await fs.writeFile(outputPath, JSON.stringify(geojson, null, 2));
  
  // 5. 一時ファイル削除
  await fs.unlink(tempPath);
  
  return {
    size: `${(Buffer.byteLength(JSON.stringify(geojson))/1024/1024).toFixed(2)} MB`,
    feature_count: geojson.features.length
  };
}

マルチサーバーフェイルオーバー


// 3台のOverpass APIサーバーでフォールバック
const servers = [
  { url: 'https://lz4.overpass-api.de/api/interpreter', host: 'lz4.overpass-api.de' },
  { url: 'https://z.overpass-api.de/api/interpreter', host: 'z.overpass-api.de' },
  { url: 'https://maps.mail.ru/osm/tools/overpass/api/interpreter', host: 'maps.mail.ru' }
];

for (const server of servers) {
  try {
    const result = await downloadToFile(
      server.url,
      fullQuery,
      output_path,
      { 'Host': server.host }  // 証明書検証用
    );
    
    return { ...result, server: server.host };
  } catch (error) {
    console.error(`Server ${server.host} failed:`, error.message);
    lastError = error;
  }
}

throw new Error(`All servers failed: ${lastError?.message}`);

2. 閾値の設定


function shouldSaveToFile(data) {
  const dataSize = JSON.stringify(data).length;
  const featureCount = data.features?.length || 0;
  
  // データサイズが1MB以上
  if (dataSize > 1024 * 1024) return true;
  
  // 地物数が1000件以上
  if (featureCount > 1000) return true;
  
  return false;
}
